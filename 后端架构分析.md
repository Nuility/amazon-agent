# 文件：agenttest/src/repositories/user_repository.py

class UserRepository:
    ENTITY_TYPE = "users"
    
    def __init__(self, storage_adapter: StorageAdapter):
        self.storage = storage_adapter
    
    def save(self, user: User) -> bool:
        return self.storage.save(self.ENTITY_TYPE, user.user_id, user.to_dict())
    
    def find_by_id(self, user_id: str) -> Optional[User]:
        data = self.storage.find_by_id(self.ENTITY_TYPE, user_id)
        if data:
            return User.from_dict(data)
        return None
    
    def find_all(self, filters: Optional[Dict] = None) -> List[User]:
        data_list = self.storage.find_all(self.ENTITY_TYPE, filters)
        return [User.from_dict(data) for data in data_list]
    
    def update(self, user: User) -> bool:
        return self.storage.update(self.ENTITY_TYPE, user.user_id, user.to_dict())
    
    def delete(self, user_id: str) -> bool:
        return self.storage.delete(self.ENTITY_TYPE, user_id)
    
    def exists(self, user_id: str) -> bool:
        return self.find_by_id(user_id) is not None
```

# 文件：agenttest/src/infrastructure/storage_adapter.py

class StorageAdapter(ABC):
    @abstractmethod
    def connect(self) -> bool:
        """建立存储连接"""
        pass
    
    @abstractmethod
    def save(self, entity_type: str, entity_id: str, data: Dict) -> bool:
        """保存实体"""
        pass
    
    @abstractmethod
    def find_by_id(self, entity_type: str, entity_id: str) -> Optional[Dict]:
        """按ID查找实体"""
        pass
    
    @abstractmethod
    def find_all(self, entity_type: str, filters: Optional[Dict]) -> List[Dict]:
        """查找所有实体（支持过滤）"""
        pass
    
    @abstractmethod
    def update(self, entity_type: str, entity_id: str, data: Dict) -> bool:
        """更新实体"""
        pass
    
    @abstractmethod
    def delete(self, entity_type: str, entity_id: str) -> bool:
        """删除实体"""
        pass
```

**文件存储实现特点**：

```python
# 文件：agenttest/src/infrastructure/file_storage_adapter.py

class FileStorageAdapter(StorageAdapter):
    def __init__(self, file_path: str, backup_path: Optional[str] = None):
        self.file_path = file_path
        self.backup_path = backup_path
        self._lock = threading.Lock()  # 线程安全
        self._data: Dict[str, Dict] = {}
    
    def connect(self) -> bool:
        # 尝试从主路径加载
        if self._load_from_path(self.file_path):
            return True
        # 主路径失败，尝试备用路径
        if self.backup_path and self._load_from_path(self.backup_path):
            return True
        return False
    
    def save(self, entity_type: str, entity_id: str, data: Dict) -> bool:
        with self._lock:
            if entity_type not in self._data:
                self._data[entity_type] = {}
            self._data[entity_type][entity_id] = data
            return self._persist()
```

**实现特性**：
- ✅ JSON格式持久化
- ✅ 线程安全（`threading.Lock`）
- ✅ 主路径失败时自动切换备用路径
- ✅ 支持过滤器查询

class Validator:
    def validate_user_data(self, user_data: Dict[str, Any]) -> None:
        # 1. 必填字段检查
        required_fields = ["username", "email"]
        for field in required_fields:
            if field not in user_data:
                raise ValidationException(field, "Field is required")
        
        # 2. 邮箱格式验证
        email = user_data["email"]
        if not self._is_valid_email(email):
            raise ValidationException("email", "Invalid email format")
        
        # 3. 用户名长度验证
        username = user_data["username"]
        if len(username) < 3 or len(username) > 50:
            raise ValidationException("username", "Username must be 3-50 characters")
        
        # 4. 手机号格式验证（可选）
        if "phone" in user_data and user_data["phone"]:
            if not self._is_valid_phone(user_data["phone"]):
                raise ValidationException("phone", "Invalid phone format")

        self.analysis_service: Optional[AnalysisService] = None

        self.amazon_ads_service: Optional[AmazonAdsService] = None
    
    def initialize(self, config_path: str = "./config/config.yaml") -> bool:
        """应用初始化流程"""
        try:
            # 1. 加载配置
            self.config_repo = ConfigRepository()
            self.config = self.config_repo.load(config_path)
            if not self.config:
                return False
            
            # 2. 初始化日志

                    backup_path="./data/users.json.bak"
                )
            else:
                raise ValueError(f"Unsupported storage type: {storage_type}")
            
            if not self.storage.connect():
                self.logger.error("Failed to connect to storage")
                return False
            
            # 4. 初始化仓储

```python
from typing import TypeVar, Generic

T = TypeVar('T')

@dataclass
class Result(Generic[T]):
    _value: Optional[T] = None
    _error: Optional[str] = None
    
    @classmethod
    def ok(cls, value: T) -> "Result[T]":
        return cls(_value=value)
    
    @classmethod
    def error(cls, error: str) -> "Result[T]":
        return cls(_error=error)
    
    def is_ok(self) -> bool:
        return self._error is None
    
    def is_error(self) -> bool:
        return self._error is not None
    
    @property
    def value(self) -> T:
        return self._value
    
    @property
    def error(self) -> str:
        return self._error
```
